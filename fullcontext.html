5) Update build.js to copy base/theme/assets per client

js
const fs = require("fs-extra");
const path = require("path");

const CLIENTS_FILE = "./data/clients.json";
const TEMPLATE_FILE = "./src/template.html";
const OUTPUT_DIR = "./dist";

const BASE_CSS = "./src/base.css";
const THEMES_DIR = "./src/themes";
const ASSETS_DIR = "./src/assets";

async function ignite() {
  console.log("üöÄ Starting Factory...");

  await fs.emptyDir(OUTPUT_DIR);

  const clients = await fs.readJson(CLIENTS_FILE);
  const template = await fs.readFile(TEMPLATE_FILE, "utf-8");

  // Copy global assets to dist once (shared)
  // If you prefer per-client assets, copy inside the loop instead.
  await fs.copy(ASSETS_DIR, path.join(OUTPUT_DIR, "assets"));

  for (const client of clients) {
    let pageContent = template;

    for (const [key, value] of Object.entries(client)) {
      const placeholder = new RegExp(`{{${key}}}`, "g");
      pageContent = pageContent.replace(placeholder, String(value));
    }

    const clientDir = path.join(OUTPUT_DIR, client.slug);
    await fs.ensureDir(clientDir);

    // Copy CSS into each client folder (so each site is fully portable)
    await fs.copy(BASE_CSS, path.join(clientDir, "base.css"));

    const themeName = client.theme || "modern";
    const themePath = path.join(THEMES_DIR, `${themeName}.css`);
    const themeExists = await fs.pathExists(themePath);

    await fs.copy(
      themeExists ? themePath : path.join(THEMES_DIR, "modern.css"),
      path.join(clientDir, "theme.css")
    );

    await fs.writeFile(path.join(clientDir, "index.html"), pageContent, "utf-8");
    console.log(`‚úÖ Generated: ${client.slug} (${themeName})`);
  }

  const indexList = clients
    .map(
      (c) =>
        `<li><a href="./${c.slug}/">${c.company_name}</a> <small>(${c.theme || "modern"})</small></li>`
    )
    .join("");

  await fs.writeFile(
    path.join(OUTPUT_DIR, "index.html"),
    `<!doctype html><meta charset="utf-8"><h1>My Client Demos</h1><ul>${indexList}</ul>`
  );

  console.log("üéâ Build Complete. Local preview: npm run preview");
}

ignite().catch((e) => {
  console.error("‚ùå Build failed:", e);
  process.exit(1);
});

4) The Bun build script (the ‚Äúfactory‚Äù)

Create build.ts:

ts
import { mkdir, rm, writeFile, cp, readdir, stat } from "node:fs/promises";
import path from "node:path";

const CLIENTS_FILE = path.join("data", "clients.json");
const TEMPLATE_FILE = path.join("src", "template.html");
const OUTPUT_DIR = "dist";

const BASE_CSS = path.join("src", "base.css");
const THEMES_DIR = path.join("src", "themes");
const ASSETS_DIR = path.join("src", "assets");

async function ensureDir(p: string) {
  await mkdir(p, { recursive: true });
}

async function emptyDir(p: string) {
  await rm(p, { recursive: true, force: true });
  await ensureDir(p);
}

// simple recursive copy (portable, no deps)
async function copyDir(from: string, to: string) {
  await ensureDir(to);
  const entries = await readdir(from);
  for (const name of entries) {
    const src = path.join(from, name);
    const dst = path.join(to, name);
    const s = await stat(src);
    if (s.isDirectory()) await copyDir(src, dst);
    else await cp(src, dst);
  }

function applyTemplate(template: string, data: Record<string, unknown>) {
  let out = template;

  }
  return out;
}
